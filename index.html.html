        // Load master dashboard
        function loadMasterDashboard() {
            // Get all booths
            const booths = [];
            for (let i = 0; i < localStorage.length; i++) {
                const key = localStorage.key(i);
                if (key.startsWith('booth_')) {
                    const booth = JSON.parse(localStorage.getItem(key));
                    booths.push(booth);
                }
            }
            
            // Calculate stats
            let totalRevenue = 0;
            let totalOrders = 0;
            
            booths.forEach(booth => {
                if (booth.salesHistory) {
                    booth.salesHistory.forEach(sale => {
                        totalRevenue += parseFloat(sale.total);
                        totalOrders++;
                    });
                }
            });
            
            document.getElementById('totalBooths').textContent = booths.length;
            document.getElementById('totalRevenue').textContent = `${totalRevenue.toFixed(2)} AED`;
            document.getElementById('totalOrders').textContent = totalOrders;
            
            // Calculate booth rankings
            const boothRankings = booths.map(booth => {
                let revenue = 0;
                let orderCount = 0;
                
                if (booth.salesHistory) {
                    booth.salesHistory.forEach(sale => {
                        revenue += parseFloat(sale.total);
                        orderCount++;
                    });
                }
                
                return {
                    boothNumber: booth.boothNumber,
                    boothName: booth.boothName,
                    revenue,
                    orderCount
                };
            }).sort((a, b) => b.revenue - a.revenue);
            
            // Render booth ranking table
            const rankingBody = document.getElementById('boothRankingBody');
            rankingBody.innerHTML = '';
            
            boothRankings.forEach((booth, index) => {
                const row = document.createElement('tr');
                row.innerHTML = `

                    <td>${booth.boothNumber}</td>
                    <td>${booth.boothName}</td>
                    <td>${booth.revenue.toFixed(2)} AED</td>

            });
        }

        // Handle submit order
        function handleSubmitOrder() {
            const customerName = document.getElementById('customerName').value;
            const orderType = document.getElementById('orderType').value;
            
            if (!customerName) {
                showNotification('Please enter customer name', 'error');
                return;
            }
            
            if (orderItems.length === 0) {
                showNotification('Please add at least one item to the order', 'error');
                return;
            }
            
            // Check stock availability
            for (const item of orderItems) {
                const stockItem = inventory.find(i => i.id === item.id);
                if (stockItem.quantity < item.quantity) {
                    showNotification(`Not enough stock for ${item.name}`, 'error');
                    return;
                }
            }
            
            // Generate order ID
            const orderId = `ORD${Date.now()}`;
            
            // Calculate total
            let total = 0;
            orderItems.forEach(item => {
                total += item.price * item.quantity;
            });
            
            // Create order object
            const order = {
                id: orderId,
                customerName,
                orderType,
                items: [...orderItems],
                total: total.toFixed(2),
                status: 'pending',
                date: new Date().toISOString(),
                staff: currentUser.username
            };
            
            // Update stock
            orderItems.forEach(item => {
                const stockItem = inventory.find(i => i.id === item.id);
                stockItem.quantity -= item.quantity;
            });
            
            // Add to orders
            orders.push(order);
            
            // Add to sales history
            salesHistory.push({
                orderId: order.id,
                date: order.date,
                items: order.items.map(item => `${item.name} x${item.quantity}`),
                total: order.total,
                staff: order.staff
            });
            
            // Save data
            currentBooth.inventory = inventory;
            currentBooth.orders = orders;
            currentBooth.salesHistory = salesHistory;
            localStorage.setItem(`booth_${currentBooth.boothNumber}`, JSON.stringify(currentBooth));
            
            // Reset form
            document.getElementById('customerName').value = '';
            orderItems = [];
            
            // Render updated data
            renderOrderItems();
            renderRecentOrders();
            renderPendingOrders();
            renderStockTable();
            renderSalesStats();
            renderSalesHistory();
            
            showNotification('Order submitted successfully!', 'success');
        }

        // Render sales stats
        function renderSalesStats() {
            const today = new Date();
            today.setHours(0, 0, 0, 0);
            
            const weekAgo = new Date(today);
            weekAgo.setDate(weekAgo.getDate() - 7);
            
            let todayRevenue = 0;
            let todayOrders = 0;
            let weekRevenue = 0;
            
            salesHistory.forEach(sale => {
                const saleDate = new Date(sale.date);
                
                if (saleDate >= today) {
                    todayRevenue += parseFloat(sale.total);
                    todayOrders++;
                }
                
                if (saleDate >= weekAgo) {
                    weekRevenue += parseFloat(sale.total);
                }
            });
            
            document.getElementById('todayRevenue').textContent = `${todayRevenue.toFixed(2)} AED`;
            document.getElementById('todayOrders').textContent = todayOrders;
            document.getElementById('weekRevenue').textContent = `${weekRevenue.toFixed(2)} AED`;
        }
